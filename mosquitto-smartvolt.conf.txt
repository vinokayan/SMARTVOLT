  <script>
    const client = mqtt.connect("ws://172.20.10.5:9001");
    const statusText = document.getElementById("status");

    client.on("connect", function () {
      statusText.innerText = "Status: MQTT Terhubung";
      console.log("Website terhubung ke MQTT");
    });

    client.on("error", function (error) {
      statusText.innerText = "Status: MQTT Error";
      console.error(error);
    });

    client.on("close", function () {
      statusText.innerText = "Status: MQTT Terputus";
    });

    function kirimPerintah(command) {
      const topic = "smartvolt/control/lampu1";

      client.publish(topic, command);

      statusText.innerText = "Status: Lampu 1 " + command;
      console.log("Kirim MQTT:", topic, command);
    }
  </script>
